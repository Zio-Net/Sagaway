# to login:
az login --use-device-code

# to see list of accounts associated
az account list --output table

# to choose which one to apply
az account set --subscription "YOUR_SUBSCRIPTION_NAME"

# Create a new Service Principal for CI/CD
az ad sp create-for-rbac \
  --name "sagaway-deployer" \
  --role Contributor \
  --scopes /subscriptions/$(az account show --query id -o tsv)/resourceGroups/SagawayDemoAKS

### output will be:
{
  "appId": "xxxx-xxxx",          # --> AZURE_CLIENT_ID
  "password": "xxxx-xxxx",       # --> AZURE_CLIENT_SECRET
  "tenant": "xxxx-xxxx",         # --> AZURE_TENANT_ID
  "subscriptionId": "xxxx-xxxx"  # --> AZURE_SUBSCRIPTION_ID (get separately)
}

# Get subscription ID
az account show --query id -o tsv


#Build your .env file:
AZURE_CLIENT_ID="your-appId"
AZURE_CLIENT_SECRET="your-password"
AZURE_TENANT_ID="your-tenant"
AZURE_SUBSCRIPTION_ID="your-subscription-id"
