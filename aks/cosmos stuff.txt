# Create the database
az cosmosdb sql database create \
  --account-name sagaway-cosmosdb \
  --resource-group SagawayDemoAKS \
  --name sagaway-db

# Create the container (state store)
MSYS_NO_PATHCONV=1 az cosmosdb sql container create \
  --account-name sagaway-cosmosdb \
  --resource-group SagawayDemoAKS \
  --database-name sagaway-db \
  --name sagaway-state \
  --partition-key-path "/partitionKey"
